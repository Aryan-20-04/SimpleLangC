
let arr = [40, 10, 20, 12, 43, 9, 86, 23];

print "Original array:";
print arr;

function Bubble(a) {
    for (let i = 0; i < length(a); i = i + 1) {
        for (let j = 0; j < length(a) - i - 1; j = j + 1) {
            if (a[j] > a[j + 1]) {
                let t = a[j];
                a[j] = a[j + 1];
                a[j + 1] = t;
            }
        }
    }
}

Bubble(arr);

print "Sorted array:";
print arr;
